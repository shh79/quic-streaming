# تبدیل به HLS
mkdir -p s3-hls/hls-stream
ffmpeg -i sample.mp4 -profile:v baseline -level 3.0 -start_number 0 \
       -hls_time 5 -hls_list_size 0 -f hls s3-hls/hls-stream/index.m3u8

# تبدیل به MPEG-DASH
mkdir -p s2-dash/dash-stream
ffmpeg -i sample.mp4 -map 0 -map 0 -c:v libx264 -b:v 800k -c:a aac -f dash s2-dash/dash-stream/video.mpd
